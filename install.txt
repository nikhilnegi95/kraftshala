clone repo from
run command "npm install"
install "db-migrate" module to system for migration using command "sudo npm install -g db-migrate"
edit "database.json" file and save db user and password
{
  "dev": {
    "driver": "mysql",
    "host": "localhost",
    "port": "3306",
    "user": "",
    "password": "",
    "database": "kraftshala"
  }
}
run command "db-migrate up" to migrate schema file
run server using command "npm start"
